10. 组件的命名空间 ​

js
// 命名空间
import Button from './components/Button.vue'

export default {
  name: 'App',
  components: {
    'my-button': {
      component: Button,
      // 命名空间
      // 命名空间可以防止组件的名称冲突
      // 也可以方便的为组件添加多个同名的实例
      // 例如：
      // <my-button></my-button>
      // <my-button></my-button>
      // <my-button></my-button>
      // 命名空间可以是任意的字符串
      // 但最好使用简短的名称
      // 例如：
      // <my-button></my-button>
      // <my-button></my-button>
      // <my-button></my-button>
      // 而不是：
      // <my-button-1></my-button-1>
      // <my-button-2></my-button-2>
      // <my-button-3></my-button-3>
      // 这样会使代码更加复杂

      namespace: 'my-button'
    }
  }
}
